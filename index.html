var data = []

var $trigger = $('[data-trigger]')
var $name = $('[data-name]')
var $postfix = $('[data-postfix]')
var $combo = $('[data-combo]').giftCombo({
    startCombo: function () {
        $name.text(`${data[0].name}送了`)
        $postfix.text('份礼物')
    },
    oneCombo: function () {
        data.shift()
        if (data.length) {
            $name.text(`${data[0].name}送了`)
        }
    },
    allCombo: function () {
        $name.text('')
        $postfix.text('')
        data.shift()
    }
})

var count = 0
$trigger.on('click', function () {
    count++
    data.push({name: `第${count}个人`, gift: count})
    $combo.send(data[data.length - 1].gift)
})
